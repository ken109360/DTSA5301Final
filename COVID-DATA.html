Part 2 Data Manipulation

US Data Manipulation

US_cases <- US_cases %>%
  pivot_longer(cols = -(UID:Combined_Key),
               names_to  = "date",
               values_to = "cases") %>%
  select(Admin2:cases) %>%
  mutate(date = mdy(date)) %>%
  select(-c(Lat, Long_))

US_deaths <- US_deaths %>%
  pivot_longer(cols = -(UID:Population),
               names_to  = "date",
               values_to = "deaths") %>%
  select(Admin2:deaths) %>%
  mutate(date = mdy(date)) %>%
  select(-c(Lat, Long_))
US <- US_cases %>%
  full_join(US_deaths)
## Joining with `by = join_by(Admin2, Province_State, Country_Region,
## Combined_Key, date)`
filtered_US <- US %>%
  filter(Province_State %in% c("Georgia", "Texas", "California", "Illinois", "Florida", "Nevada", "North Carolina"))
filtered_US_agg <- filtered_US %>%
  group_by(date) %>%
  summarize(total_cases = sum(cases, na.rm = TRUE))

# Extract month from the date and aggregate cases by month
global_monthly <- global %>%
  mutate(month = format(date, "%Y-%m")) %>%  # Extract year-month
  group_by(month) %>%
  summarize(total_cases = sum(cases, na.rm = TRUE))
